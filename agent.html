最后一步：定时心跳 SQL
开启 pg_cron 和 pg_net 扩展后，在 SQL Editor 运行：
-- 清理旧任务
select cron.unschedule('my-agent-tick');

-- 每 15 分钟唤醒 Edge Function
select cron.schedule(
  'my-agent-tick','*/15 * * * *',
  $$select net.http_post(
    url:='YOUR_EDGE_URL',
    headers:='{"Content-Type":"application/json","Authorization":"Bearer YOUR_ANON_KEY"}'::jsonb
  )$$
);
